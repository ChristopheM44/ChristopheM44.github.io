    <h1>Scores Flip7</h1>

        <button class="btn-add" onclick="addPlayer()">Ajouter</button>

    </div>


    <div id="playersList">

        </div>


    <script>

        // Clé utilisée pour sauvegarder dans le stockage local du navigateur

        const STORAGE_KEY = 'flip7_scores_data';

        

        // Chargement des données existantes ou initialisation d'un tableau vide

        let players = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];


        // Fonction pour sauvegarder les données

        function savePlayers() {

            localStorage.setItem(STORAGE_KEY, JSON.stringify(players));

        }


        // Fonction pour afficher les joueurs à l'écran

        function renderPlayers() {

            const listContainer = document.getElementById('playersList');

            listContainer.innerHTML = ''; // On vide la liste avant de la recréer


            players.forEach((player, index) => {

                const playerEl = document.createElement('div');

                playerEl.className = 'player-card';

                

                playerEl.innerHTML = `

                    <div class="player-info">

                        <p class="player-name">${player.name}</p>

                        <p class="player-score">${player.score}</p>

                    </div>

                    <div class="score-controls">

                        <button class="btn-score" onclick="updateScore(${index}, -1)">-</button>

                        <button class="btn-score" onclick="updateScore(${index}, 1)">+</button>

                        <button class="btn-delete" onclick="removePlayer(${index})">✕</button>

                    </div>

                `;

                listContainer.appendChild(playerEl);

            });

        }


        // Ajouter un joueur

        function addPlayer() {

            const input = document.getElementById('playerNameInput');

            const name = input.value.trim();

            

            if (name !== "") {

                players.push({ name: name, score: 0 });

                input.value = ''; // On vide le champ

                savePlayers();

                renderPlayers();

            }

        }


        // Permet d'ajouter un joueur en appuyant sur "Entrée" sur le clavier

        function handleKeyPress(event) {

            if (event.key === 'Enter') {

                addPlayer();

            }

        }


        // Modifier le score

        function updateScore(index, change) {

            players[index].score += change;

            savePlayers();

            renderPlayers();

        }


        // Supprimer un joueur

        function removePlayer(index) {

            if (confirm("Voulez-vous vraiment supprimer ce joueur ?")) {

                players.splice(index, 1);

                savePlayers();

                renderPlayers();

            }

        }


        // Affichage initial au chargement de la page

        renderPlayers();
